# Data preprocess
data = pd.read_csv('/content/drive/MyDrive/Sleep_health_and_lifestyle_dataset.csv')

selected_cols = ['Gender', 'Age', 'Occupation', 'Sleep Duration', 'Quality of Sleep',
                 'Physical Activity Level', 'Stress Level', 'BMI Category', 'Heart Rate', 'Daily Steps',
                 'Sleep Disorder', 'Blood Pressure']

df = data[selected_cols].copy()

label_encoder_gender = LabelEncoder()
df['Gender'] = label_encoder_gender.fit_transform(df['Gender'])

occupation_mapping = {
    'Software Engineer': 0,
    'Doctor': 1,
    'Sales Representative': 2,
    'Salesperson': 2,
    'Teacher': 3,
    'Nurse': 4,
    'Engineer': 5,
    'Accountant': 6,
    'Scientist': 7,
    'Lawyer': 8,
    'Manager': 9
}
df['Occupation'] = df['Occupation'].map(occupation_mapping)

bmi_mapping = {
    'Overweight': 1,
    'Normal': 0,
    'Obese': 2,
    'Normal Weight': 0
}
df['BMI Category'] = df['BMI Category'].map(bmi_mapping)

df['UnderPressure'] = df['Blood Pressure'].apply(lambda x: int(x.split("/")[1]))
df['OverPressure'] = df['Blood Pressure'].apply(lambda x: int(x.split("/")[0]))
df = df.drop('Blood Pressure', axis=1)

sleep_mapping = {
    'None': 0,
    'Sleep Apnea': 1,
    'Insomnia': 2
}
df['Sleep Disorder'] = df['Sleep Disorder'].map(sleep_mapping)

features = ['Gender', 'Age', 'Occupation', 'Sleep Duration', 'Quality of Sleep',
            'Physical Activity Level', 'Stress Level', 'BMI Category', 'Heart Rate', 'Daily Steps',
            'UnderPressure', 'OverPressure', 'Sleep Disorder']

X = df[features]
y = df['Sleep Disorder']

# One-hot encoding of the categorical columns
categorical_cols = ['Gender', 'Occupation', 'BMI Category']
X = pd.get_dummies(X, columns=categorical_cols, drop_first=True)

X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, stratify=y, random_state=42)

scaler = StandardScaler()
X_train = scaler.fit_transform(X_train)
X_test = scaler.transform(X_test)

# Create model, training and prediction
model = keras.Sequential([
    layers.Dense(64, activation='relu', input_shape=(X_train.shape[1],)),
    layers.BatchNormalization(),
    layers.Dropout(0.1),
    layers.Dense(32, activation='relu'),
    layers.BatchNormalization(),
    layers.Dropout(0.1),
    layers.Dense(3, activation='softmax')
])

optimizer = keras.optimizers.Adam(learning_rate=0.01)
model.compile(optimizer=optimizer, loss='sparse_categorical_crossentropy', metrics=['accuracy'])
res = model.fit(X_train, y_train, epochs=100, batch_size=32, validation_data=(X_test, y_test), verbose=0)

_, accuracy = model.evaluate(X_test, y_test, verbose=0)
print(f'Accuracy on test data: {accuracy * 100:.2f}%')

# Confusion matrix

def plot_conf_matrix(model, X_test, y_test):
    y_pred = model.predict(X_test)
    y_pred_classes = np.argmax(y_pred, axis=1)
    cm = confusion_matrix(y_test, y_pred_classes)
    plt.figure(figsize=(8, 6))
    sns.heatmap(cm, annot=True, fmt='d', cmap='Blues', xticklabels=['Class 0', 'Class 1', 'Class 2'], yticklabels=['Class 0', 'Class 1', 'Class 2'])
    plt.title('Confusion matrix')
    plt.xlabel('Predicted')
    plt.ylabel('Actual')
    plt.show()

plot_conf_matrix(model, X_test, y_test)
